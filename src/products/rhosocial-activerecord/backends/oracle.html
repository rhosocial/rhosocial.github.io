Connection

连接配置。

OracleConnectionConfig 继承 ConnectionConfig，混入 Pool / SSL / Timezone / Version / Logging 五组配置。

connection_config.py
from rhosocial.activerecord.backend.impl.oracle import OracleBackend
from rhosocial.activerecord.backend.impl.oracle.config import OracleConnectionConfig

# 标准连接（oracledb 驱动）
config = OracleConnectionConfig(
    host='localhost',
    port=1521,
    user='app_user',
    password='secret',
    service='ORCL',
)
backend = OracleBackend(connection_config=config)
backend.connect()

# 快捷方式：直接传连接参数
backend = OracleBackend(
    host='localhost',
    port=1521,
    user='app_user',
    password='secret',
    service='ORCL',
)

# Oracle 专有字段
config = OracleConnectionConfig(
    ...,
    mode='sysdba',
    thread_mode=True,
    events=False,
    ssl_server_dn_match=True,
    config_dir='/opt/oracle/network/admin',
)

# 从环境变量创建配置（前缀 ORACLE_）
config = OracleConnectionConfig.from_env(prefix="ORACLE_")

Dialect

OracleDialect 版本感知。

OracleDialect 依赖 _ORACLE_VERSION_BOUNDARIES 字典，版本阈值驱动功能开关：10g / 11g / 12c / 12.2 / 18c / 21c / 23ai。

dialect_version.py
from rhosocial.activerecord.backend.impl.oracle.dialect import OracleDialect

# 默认版本 23.0.0（启用全部功能）
dialect = OracleDialect()

# 指定旧版本 → 功能降级
old_dialect = OracleDialect(version=(11, 0, 0))
old_dialect.supports_fetch_first_pagination() # False (需 12c+)
old_dialect.supports_pivot()                      # True (11g+)
old_dialect.supports_json_type()                   # False (需 21c+)

# 版本阈值表 (_ORACLE_VERSION_BOUNDARIES)
# 10g — MERGE / RETURNING / Flashback Query
# 11g — PIVOT / UNPIVOT / LISTAGG / READ ONLY
# 12c — FETCH FIRST / IDENTITY 列 / LATERAL
# 12.2 — JSON 函数
# 18c — POLYMORPHIC TABLE FUNCTION
# 21c — 原生 JSON / BITMAP INDEX
# 23ai — BOOLEAN 类型 / VECTOR 类型 / JSON DUALITY

Hierarchical

层次查询示例。

hierarchical_query.py
from rhosocial.activerecord.backend.impl.oracle.dialect import OracleDialect

dialect = OracleDialect()

# 员工层级结构查询
sql = dialect.format_connect_by(
    table='employees',
    connect_condition='PRIOR employee_id = manager_id',
    start_with='manager_id IS NULL',
    order_siblings_by='last_name',
)
# → SELECT ... FROM employees
#   START WITH manager_id IS NULL
#   CONNECT BY PRIOR employee_id = manager_id
#   ORDER SIBLINGS BY last_name

# 层次函数
sql = dialect.format_connect_by_root(column='last_name')
# → CONNECT_BY_ROOT last_name

sql = dialect.format_sys_connect_by_path(column='last_name', separator='/')
# → SYS_CONNECT_BY_PATH(last_name, '/')

# NOCYCLE 防止循环
# CONNECT BY NOCYCLE PRIOR employee_id = manager_id
PIVOT

PIVOT / UNPIVOT 示例。

pivot_example.py
from rhosocial.activerecord.backend.impl.oracle.dialect import OracleDialect

dialect = OracleDialect()

# PIVOT：行转列
sql, params = dialect.format_pivot_clause(
    aggregate='SUM',
    pivot_column='quarter',
    value_column='revenue',
    pivot_values=['Q1', 'Q2', 'Q3', 'Q4'],
)
# → PIVOT (SUM(revenue) FOR quarter IN ('Q1', 'Q2', 'Q3', 'Q4'))

# UNPIVOT：列转行
sql, params = dialect.format_unpivot_clause(
    value_column='revenue',
    pivot_column='quarter',
    unpivot_columns=['Q1', 'Q2', 'Q3', 'Q4'],
    include_nulls=False,
)
# → UNPIVOT EXCLUDE NULLS (revenue FOR quarter IN (Q1, Q2, Q3, Q4))
MERGE

MERGE 语句示例。

merge_example.py
from rhosocial.activerecord.backend.impl.oracle.dialect import OracleDialect

dialect = OracleDialect()

# MERGE INTO：条件插入或更新
sql, params = dialect.format_merge_statement(
    target='employees',
    source='(SELECT :emp_id AS employee_id, :name AS name FROM DUAL)',
    on_condition='employees.employee_id = source.employee_id',
    when_matched={'update': {'name': 'source.name'}},
    when_not_matched={'insert': {'employee_id': 'source.employee_id', 'name': 'source.name'}},
)
# → MERGE INTO employees
#   USING (SELECT :emp_id AS employee_id, :name AS name FROM DUAL) source
#   ON (employees.employee_id = source.employee_id)
#   WHEN MATCHED THEN UPDATE SET name = source.name
#   WHEN NOT MATCHED THEN INSERT (employee_id, name) VALUES (source.employee_id, source.name)

# UPSERT 通过 MERGE 实现
# OracleDialect.get_upsert_syntax_type() → "MERGE"
Hints

查询提示示例。

hint_example.py
from rhosocial.activerecord.backend.impl.oracle.dialect import OracleDialect

dialect = OracleDialect()

# 单提示
hint = dialect.format_query_hint(hints=['FULL'], table='employees')
# → /*+ FULL(employees) */

# 多提示组合
hint = dialect.format_query_hint(hints=['PARALLEL', 'USE_HASH'], table='orders', degree=4)
# → /*+ PARALLEL(orders 4) USE_HASH(orders) */

# 常用提示类型
# FULL(table)         — 全表扫描
# INDEX(table idx)    — 索引扫描
# PARALLEL(table n)   — 并行执行
# USE_NL(table)       — 嵌套循环连接
# USE_HASH(table)     — 哈希连接
# DRIVING_SITE(table) — 分布式查询远程执行
# APPEND              — 直接路径插入
# FIRST_ROWS(n)       — 优化前 n 行响应
# ALL_ROWS            — 优化全量吞吐

# Flashback Query：历史时点查询
sql = dialect.format_flashback_query_as_of(
    table='orders',
    timestamp='2024-01-01 00:00:00',
)
# → SELECT * FROM orders AS OF TIMESTAMP('2024-01-01 00:00:00')

Transaction

事务控制。

OracleTransactionManager 支持四种隔离级别，默认 READ_COMMITTED。支持 SET TRANSACTION / SAVEPOINT / 只读事务。

transaction_control.py
from rhosocial.activerecord.backend.transaction import IsolationLevel

# 标准事务（默认 READ COMMITTED）
with backend.transaction():
    backend.execute("INSERT INTO employees ...", ())

# 指定隔离级别
with backend.transaction(isolation_level=IsolationLevel.SERIALIZABLE):
    ...

# Oracle 四种隔离级别
# READ_UNCOMMITTED (实际等同于 READ_COMMITTED)
# READ_COMMITTED  (默认)
# REPEATABLE_READ (实际等同于 SERIALIZABLE)
# SERIALIZABLE

# 只读事务
with backend.transaction(read_only=True):
    result = backend.execute("SELECT ...", ())

# Savepoint 支持
with backend.transaction():
    backend.execute("INSERT ...", ())
    with backend.transaction():  # SAVEPOINT
        backend.execute("UPDATE ...", ())
Async

异步支持。

AsyncOracleBackend 懒加载，使用 oracledb 异步驱动。

async_oracle.py
from rhosocial.activerecord.backend.impl.oracle import AsyncOracleBackend
# 需安装: pip install rhosocial-activerecord-oracle[async]

backend = AsyncOracleBackend(
    host='localhost',
    port=1521,
    user='app_user',
    password='secret',
    service='ORCL',
)
await backend.connect()

async with backend.transaction():
    result = await backend.execute("SELECT ...", ())

await backend.disconnect()

# 懒加载：同步组件不依赖异步驱动
from rhosocial.activerecord.backend.impl.oracle import OracleBackend       # ✓
from rhosocial.activerecord.backend.impl.oracle import AsyncOracleBackend # ✗ 需要异步驱动
