SQL Server

微软企业级 RDBMS。OUTPUT 子句 / MERGE / CROSS APPLY / IDENTITY / OFFSET-FETCH / 表提示 / 索引视图。

Features

SQL Server 特性。

01 / OUTPUT

OUTPUT 子句

INSERT / UPDATE / DELETE / MERGE 支持 OUTPUT 子句。inserted.* / deleted.* 虚拟表，替代 RETURNING。

02 / MERGE

MERGE

MERGE ... USING ... ON ... WHEN MATCHED / NOT MATCHED。支持 $action 输出。通过 MERGE 实现 UPSERT。

03 / APPLY

CROSS / OUTER APPLY

替代 LATERAL JOIN。CROSS APPLY（内连接） / OUTER APPLY（左连接）。关联子查询横向关联。

04 / IDENTITY

IDENTITY

IDENTITY(1,1) 自增列。SET IDENTITY_INSERT ON/OFF。SCOPE_IDENTITY() / IDENT_CURRENT() / DBCC CHECKIDENT。

05 / Pagination

OFFSET-FETCH

OFFSET n ROWS FETCH NEXT m ROWS ONLY（2012+）。旧版本使用 TOP / ROW_NUMBER() 分页。

06 / Hints

表提示

WITH (NOLOCK / UPDLOCK / XLOCK / READPAST / TABLOCKX / INDEX / HOLDLOCK) — 查询级锁控制。

Connection

连接配置。

SQLServerConnectionConfig 继承 pydantic.BaseModel，纯 Pydantic 模型（无框架混入）。ODBC 连接字符串构建。

connection_config.py
from rhosocial.activerecord.backend.impl.sqlserver import SQLServerBackend
from rhosocial.activerecord.backend.impl.sqlserver.config import SQLServerConnectionConfig

# 标准连接（pyodbc 驱动）
config = SQLServerConnectionConfig(
    host='localhost',
    port=1433,
    database='app_db',
    username='sa',
    password='YourPassword123',
)
backend = SQLServerBackend(connection_config=config)
backend.connect()

# Windows 身份验证（Trusted Connection）
config = SQLServerConnectionConfig(
    host='localhost',
    port=1433,
    database='app_db',
    trusted_connection=True,
)

# ODBC 驱动选择
config = SQLServerConnectionConfig(
    ...,
    driver='ODBC Driver 18 for SQL Server',
    encrypt=True,
    trust_server_certificate=True,
)

# 连接字符串自动构建
conn_str = config.build_connection_string()
# → DRIVER={ODBC Driver 17 for SQL Server};SERVER=localhost;DATABASE=app_db;...

Dialect

SQLServerDialect 版本感知。

SQLServerDialect 依赖版本常量：2012 / 2014 / 2016 / 2017 / 2019 / 2022，驱动功能开关。

dialect_version.py
from rhosocial.activerecord.backend.impl.sqlserver.dialect import SQLServerDialect

# 默认版本 16.0.0 (SQL Server 2022)
dialect = SQLServerDialect()

# 指定旧版本 → 功能降级

old_dialect.supports_json_type()                   # False (需 2016+)
old_dialect.supports_fetch_first_pagination()      # True (2012+)
old_dialect.supports_for_update_skip_locked()      # False (需 2019+)

# 版本常量 (SQL_SERVER_* = (major, 0, 0))
# SQL_SERVER_2012 = (11, 0, 0) — OFFSET FETCH / SEQUENCE / IIF / TRY_CAST / Window Frame
# SQL_SERVER_2014 = (12, 0, 0)
# SQL_SERVER_2016 = (13, 0, 0) — JSON / STRING_SPLIT / Temporal Tables / DROP IF EXISTS
# SQL_SERVER_2017 = (14, 0, 0) — STRING_AGG / CONCAT_WS / TRIM
# SQL_SERVER_2019 = (15, 0, 0) — READPAST+UPDLOCK / APPROX_COUNT_DISTINCT
# SQL_SERVER_2022 = (16, 0, 0) — JSON_PATH_EXISTS / JSON_OBJECT / JSON_ARRAY

OUTPUT

OUTPUT 子句示例。

output_example.py
from rhosocial.activerecord.backend.impl.sqlserver.dialect import SQLServerDialect

dialect = SQLServerDialect()

# INSERT ... OUTPUT — 替代 RETURNING
sql, params = dialect.format_insert_statement(
    table='users',
    columns=['id', 'name'],
    values=[[1, 'Alice']],
    output_columns=['id', 'name'],
)
# → INSERT INTO users (id, name) OUTPUT inserted.id, inserted.name VALUES (1, 'Alice')

# UPDATE ... OUTPUT — 同时获取旧值和新值
sql, params = dialect.format_update_statement(
    table='users',
    set_clause={'name': 'Bob'},
    where='id = 1',
    output_columns=['deleted.name', 'inserted.name'],
)
# → UPDATE users SET name = 'Bob' OUTPUT deleted.name, inserted.name WHERE id = 1

# DELETE ... OUTPUT
sql, params = dialect.format_delete_statement(
    table='users',
    where='id = 1',
    output_columns=['deleted.id', 'deleted.name'],
)
# → DELETE FROM users OUTPUT deleted.id, deleted.name WHERE id = 1

# MERGE ... OUTPUT $action — 区分 INSERT / UPDATE
# OUTPUT $action, inserted.*, deleted.*
MERGE

MERGE 语句示例。

merge_example.py
from rhosocial.activerecord.backend.impl.sqlserver.dialect import SQLServerDialect

dialect = SQLServerDialect()

# MERGE INTO：条件插入或更新（SQL Server UPSERT 机制）
# SQLServerDialect.get_upsert_syntax_type() → "MERGE"

# MERGE with OUTPUT $action
# MERGE INTO users AS target
# USING (SELECT @id AS id, @name AS name) AS source
# ON target.id = source.id
# WHEN MATCHED THEN UPDATE SET name = source.name
# WHEN NOT MATCHED THEN INSERT (id, name) VALUES (source.id, source.name)
# OUTPUT $action, inserted.*, deleted.*;

# $action 输出列值：
#   'INSERT' — 新插入行
#   'UPDATE' — 已更新行
#   'DELETE' — 仅 DELETE WHEN MATCHED 时
SET

SET 语句控制。

set_statements.py
from rhosocial.activerecord.backend.impl.sqlserver.mixins import SQLServerBackendMixin

# SET NOCOUNT — 禁止返回行数消息
backend.set_nocount(on=True)   # → SET NOCOUNT ON

# IDENTITY 控制
backend.set_identity_insert(table='users', on=True)
# → SET IDENTITY_INSERT users ON

identity = backend.get_scope_identity()
# → SELECT SCOPE_IDENTITY()

identity = backend.get_identity_current(table='users')
# → SELECT IDENT_CURRENT('users')

backend.reset_identity(table='users', seed=1000)
# → DBCC CHECKIDENT ('users', RESEED, 1000)

# 锁和死锁控制
backend.set_lock_timeout(timeout_ms=5000)       # → SET LOCK_TIMEOUT 5000
backend.set_deadlock_priority(priority=10)        # → SET DEADLOCK_PRIORITY 10

# 会话设置
backend.set_ansi_nulls(on=True)              # → SET ANSI_NULLS ON
backend.set_quoted_identifier(on=True)       # → SET QUOTED_IDENTIFIER ON
backend.set_ansi_padding(on=True)            # → SET ANSI_PADDING ON
backend.set_textsize(size=2147483647)        # → SET TEXTSIZE 2147483647
backend.set_language(language='us_english')     # → SET LANGUAGE us_english
backend.set_dateformat(format='ymd')           # → SET DATEFORMAT ymd

# 服务器属性查询
version = backend.get_product_version()    # → SERVERPROPERTY('ProductVersion')
edition  = backend.get_edition()           # → SERVERPROPERTY('Edition')

Transaction

事务控制。

SQLServerTransactionManager 支持五种隔离级别，含 SQL Server 专有 SNAPSHOT。支持 SET LOCK_TIMEOUT / DEADLOCK_PRIORITY。

transaction_control.py
from rhosocial.activerecord.backend.transaction import IsolationLevel

# 标准事务（默认 READ COMMITTED）
with backend.transaction():
    backend.execute("INSERT INTO users ...", ())

# 指定隔离级别
with backend.transaction(isolation_level=IsolationLevel.SERIALIZABLE):
    ...

# SQL Server 五种隔离级别
# READ_UNCOMMITTED
# READ_COMMITTED  (默认)
# REPEATABLE_READ
# SERIALIZABLE
# SNAPSHOT — SQL Server 专有！

# SNAPSHOT 隔离级别（行版本控制）
with backend.transaction(snapshot=True):
    result = backend.execute("SELECT ...", ())

# Savepoint 支持
with backend.transaction():
    backend.execute("INSERT ...", ())
    with backend.transaction():  # SAVE TRANSACTION
        backend.execute("UPDATE ...", ())

# 注意：SQL Server 中 ROLLBACK 回滚所有嵌套层
# 嵌套事务使用 SQLServerNestedTransactionManager

# 锁控制
backend.set_lock_timeout(5000)      # SET LOCK_TIMEOUT 5000
backend.set_deadlock_priority(10)  # SET DEADLOCK_PRIORITY 10
Async

异步支持。

AsyncSQLServerBackend 懒加载，使用 aioodbc 异步驱动。注意：继承同步 StorageBackend 基类。

async_sqlserver.py
from rhosocial.activerecord.backend.impl.sqlserver import AsyncSQLServerBackend
# 需安装: pip install rhosocial-activerecord-sqlserver[async]

backend = AsyncSQLServerBackend(
    host='localhost',
    port=1433,
    database='app_db',
    username='sa',
    password='YourPassword123',
)
await backend.connect()

async with backend.transaction():
    result = await backend.execute("SELECT ...", ())

await backend.disconnect()

# 懒加载：同步组件不依赖异步驱动
from rhosocial.activerecord.backend.impl.sqlserver import SQLServerBackend         # ✓
from rhosocial.activerecord.backend.impl.sqlserver import AsyncSQLServerBackend # ✗ 需要异步驱动
