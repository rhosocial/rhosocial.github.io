MariaDB

MySQL 兼容的增强分支。SEQUENCE / RETURNING / 系统版本化表 / INTERSECT & EXCEPT 独有支持。

Features

MariaDB 特性。

01 / SEQUENCE

序列

CREATE SEQUENCE / NEXTVAL / CURRVAL / SETVAL（10.3+）。独立序列对象，替代 AUTO_INCREMENT。

02 / RETURNING

RETURNING

INSERT / DELETE / REPLACE 支持 RETURNING 子句（10.5+）。MySQL 不支持此功能。

03 / Versioning

系统版本化

WITH SYSTEM VERSIONING / FOR SYSTEM_TIME AS OF / BETWEEN（10.3+）。时点查询，无需额外表。

04 / SET Ops

集合运算

INTERSECT / EXCEPT / INTERSECT ALL / EXCEPT ALL（10.3+）。MySQL 8.0.31+ 才部分支持。

Connection

连接配置。

MariaDBConnectionConfig 继承 ConnectionConfig，混入 Pool / SSL / Charset / Timezone / Version / Logging 六组配置。

connection_config.py
from rhosocial.activerecord.backend.impl.mariadb import MariaDBBackend
from rhosocial.activerecord.backend.impl.mariadb.config import MariaDBConnectionConfig

# 标准连接（使用官方 mariadb 驱动 + 连接池）
config = MariaDBConnectionConfig(
    host='127.0.0.1',
    port=3306,
    user='app_user',
    password='secret',
    database='app_db',
)
backend = MariaDBBackend(connection_config=config)
backend.connect()

# 快捷方式：直接传连接参数
backend = MariaDBBackend(
    host='127.0.0.1',
    port=3306,
    user='app_user',
    password='secret',
    database='app_db',
)

# MariaDB 专有字段
config = MariaDBConnectionConfig(
    ...,
    autocommit=False,
    ssl_disabled=False,
    options={'plugindir': '/usr/lib/mariadb/plugin'},
)

# 从环境变量创建配置（前缀 MARIADB_）
config = MariaDBConnectionConfig.from_env(prefix="MARIADB_")

Dialect

MariaDBDialect 版本感知。

MariaDBDialect 依赖 MARIADB_VERSION_BOUNDARIES 字典，11 个版本阈值驱动功能开关。

dialect_version.py
from rhosocial.activerecord.backend.impl.mariadb.dialect import MariaDBDialect

# 默认版本 10.11.0（启用全部功能）
dialect = MariaDBDialect()

# 指定旧版本 → 功能降级
old_dialect = MariaDBDialect(version=(10, 2, 0))
old_dialect.supports_returning_clause() # False (需 10.5+)
old_dialect.supports_sequence() # False (需 10.3+)
old_dialect.supports_window_functions() # True (10.2+)

# 版本阈值表 (MARIADB_VERSION_BOUNDARIES)
# 10.2.0 — Window Functions / CTE / FILTER
# 10.2.3 — JSON 函数
# 10.2.7 — JSON 箭头操作符 (-> / ->>)
# 10.3.0 — SEQUENCE / System Versioning / INTERSECT / EXCEPT / SKIP LOCKED
# 10.4.0 — INSTEAD OF 触发器 / 每事件多触发器
# 10.5.0 — RETURNING 子句
# 10.6.0 — EXPLAIN FORMAT=JSON/TREE / EXPLAIN ANALYZE

Sequence

序列操作示例。

sequence_example.py
from rhosocial.activerecord.backend.impl.mariadb.dialect import MariaDBDialect

dialect = MariaDBDialect()

# CREATE SEQUENCE
sql, params = dialect.format_create_sequence_statement(
    name='user_id_seq',
    start_with=1,
    increment_by=1,
    minvalue=1,
    maxvalue=999999,
    cache=20,
    cycle=False,
)
# → CREATE SEQUENCE `user_id_seq` START WITH = 1 INCREMENT BY = 1 ...

# NEXTVAL / CURRVAL / SETVAL
sql = dialect.format_nextval(sequence_name='user_id_seq')
# → NEXT VALUE FOR `user_id_seq`

sql = dialect.format_currval(sequence_name='user_id_seq')
# → CURRENT VALUE FOR `user_id_seq`

# ALTER SEQUENCE
sql, params = dialect.format_alter_sequence_statement(
    name='user_id_seq',
    restart_with=1000,
)

# DROP SEQUENCE
sql, params = dialect.format_drop_sequence_statement(name='user_id_seq')
Versioning

系统版本化示例。

versioning_example.py
from rhosocial.activerecord.backend.impl.mariadb.dialect import MariaDBDialect

dialect = MariaDBDialect()

# 建表时启用系统版本化
sql, params = dialect.format_system_versioning_clause()
# → WITH SYSTEM VERSIONING

# 时点查询：AS OF
sql, params = dialect.format_for_system_time_as_of(
    table_name='users',
    timestamp='2024-01-01 00:00:00'
)
# → `users` FOR SYSTEM_TIME AS OF '2024-01-01 00:00:00'

# 区间查询：BETWEEN
sql, params = dialect.format_for_system_time_between(
    table_name='users',
    start='2024-01-01',
    end='2024-06-01'
)
# → `users` FOR SYSTEM_TIME BETWEEN '2024-01-01' AND '2024-06-01'

# 全历史查询
sql, params = dialect.format_for_system_time_all(table_name='users')
# → `users` FOR SYSTEM_TIME ALL
INSERT Syntax

INSERT 语法扩展。

insert_syntax.py
from rhosocial.activerecord.backend.expression import InsertExpression, ValuesSource, Literal

# ON DUPLICATE KEY UPDATE（与 MySQL 相同）
insert = InsertExpression(
    dialect=dialect, into='users',
    columns=['id', 'name'],
    source=ValuesSource(dialect, [[Literal(dialect, 1), Literal(dialect, 'Alice')]]),
    dialect_options={'on_duplicate_key_update': {'name': Literal(dialect, 'Alice')}}
)

# INSERT ... RETURNING（MariaDB 10.5+ 独有，MySQL 不支持）
insert_returning = InsertExpression(
    dialect=dialect, into='users',
    columns=['id', 'name'],
    source=ValuesSource(dialect, [[Literal(dialect, 1), Literal(dialect, 'Alice')]]),
    dialect_options={'returning': ['id', 'name']}
)
# → INSERT INTO users (id, name) VALUES (1, 'Alice') RETURNING id, name

# REPLACE INTO ... RETURNING（MariaDB 10.5+ 独有）
replace = InsertExpression(
    dialect=dialect, into='users',
    columns=['id', 'name'],
    source=ValuesSource(dialect, [[Literal(dialect, 1), Literal(dialect, 'Alice')]]),
    dialect_options={'replace': True, 'returning': ['id']}
)

# INSERT IGNORE（与 MySQL 相同）
ignore = InsertExpression(
    dialect=dialect, into='users',
    columns=['id', 'name'],
    source=ValuesSource(dialect, [[Literal(dialect, 1), Literal(dialect, 'Alice')]]),
    dialect_options={'ignore': True}
)

Transaction

事务控制。

MariaDBTransactionManager 支持全部四种 SQL 标准隔离级别。默认 REPEATABLE_READ。支持 START TRANSACTION READ ONLY。

transaction_control.py
from rhosocial.activerecord.backend.transaction import IsolationLevel

# 标准事务（默认 REPEATABLE READ）
with backend.transaction():
    backend.execute("INSERT INTO users ...", ())

# 指定隔离级别
with backend.transaction(isolation_level=IsolationLevel.SERIALIZABLE):
    ...

# MariaDB 四种隔离级别
# READ_UNCOMMITTED / READ_COMMITTED / REPEATABLE_READ / SERIALIZABLE

# MariaDB 独有：START TRANSACTION READ ONLY（10.0+）
dialect = backend.dialect
sql, params = dialect.format_start_transaction(read_only=True)
# → START TRANSACTION READ ONLY

# FOR UPDATE SKIP LOCKED / NOWAIT（10.3+）
# 通过查询表达式的 dialect_options 设置

# Savepoint 支持
with backend.transaction():
    backend.execute("INSERT ...", ())
    with backend.transaction(): # SAVEPOINT
        backend.execute("UPDATE ...", ())
Async

异步支持。

AsyncMariaDBBackend 懒加载。优先使用 mariadb.aio，回退到 aiomysql。

async_mariadb.py
from rhosocial.activerecord.backend.impl.mariadb import AsyncMariaDBBackend
# 需安装: pip install rhosocial-activerecord-mariadb[async]

backend = AsyncMariaDBBackend(
    host='127.0.0.1',
    port=3306,
    user='app_user',
    password='secret',
    database='app_db',
)
await backend.connect()

async with backend.transaction():
    result = await backend.execute("SELECT ...", ())

await backend.disconnect()

# 驱动优先级：mariadb.aio → aiomysql (回退)
# 懒加载：导入同步组件不需要异步驱动
from rhosocial.activerecord.backend.impl.mariadb import MariaDBBackend # ✓
from rhosocial.activerecord.backend.impl.mariadb import AsyncMariaDBBackend # ✗ 需要异步驱动
