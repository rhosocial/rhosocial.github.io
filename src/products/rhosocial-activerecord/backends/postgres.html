Connection

连接配置。

PostgresConnectionConfig 继承 ConnectionConfig，混入 SSL / SearchPath / Timezone 三组配置。

connection_config.py
from rhosocial.activerecord.backend.impl.postgresql import PostgresBackend
from rhosocial.activerecord.backend.impl.postgresql.config import PostgresConnectionConfig

# 标准连接
config = PostgresConnectionConfig(
    host='127.0.0.1',
    port=5432,
    user='app_user',
    password='secret',
    database='app_db',
)
backend = PostgresBackend(connection_config=config)
backend.connect()

# 快捷方式：直接传连接参数
backend = PostgresBackend(
    host='127.0.0.1',
    port=5432,
    user='app_user',
    password='secret',
    database='app_db',
)

# 从环境变量创建配置（前缀 PG_）
config = PostgresConnectionConfig.from_env(prefix="PG_")

    timezone='Asia/Shanghai',
)

ENUM DDL

枚举类型 DDL。

enum_ddl.py
from rhosocial.activerecord.backend.impl.postgresql.dialect import PostgresDialect
from rhosocial.activerecord.backend.impl.postgresql.type import PostgresEnumType

dialect = PostgresDialect()

# 创建枚举类型
enum_type = PostgresEnumType(name='user_status', values=['active', 'inactive', 'suspended'])

# CREATE TYPE
sql, params = dialect.format_create_enum_type(
    type_name='user_status',
    values=['active', 'inactive', 'suspended']
)
# → CREATE TYPE "user_status" AS ENUM ('active', 'inactive', 'suspended')

# ALTER TYPE ADD VALUE（PostgreSQL 12+ 支持事务内添加）
sql, params = dialect.format_alter_enum_type_add_value(
    type_name='user_status',
    new_value='banned'
)
# → ALTER TYPE "user_status" ADD VALUE 'banned'

# DROP TYPE
sql, params = dialect.format_drop_enum_type(type_name='user_status')
# → DROP TYPE IF EXISTS "user_status"
Dialect

PostgresDialect 版本感知。

PostgresDialect 在初始化时接受版本元组，动态启用/禁用协议方法。

dialect_version.py
from rhosocial.activerecord.backend.impl.postgresql.dialect import PostgresDialect

# 默认版本 15.0（启用 MERGE）
dialect = PostgresDialect()

# 指定旧版本 → 功能降级
old_dialect = PostgresDialect(version=(11, 0, 0))
old_dialect.supports_merge() # False
old_dialect.supports_returning_clause() # True
old_dialect.supports_partition() # True (10+)

# 版本对应功能表
# 10 — 声明式分区
# 12 — CTE 物化控制 (NOT MATERIALIZED)
# 13 — 生成列 (GENERATED ALWAYS AS)
# 14 — UNBOUNDED FOLLOWING 窗口帧
# 15 — MERGE 语句

DDL Example

DDL 操作示例。

ddl_example.py
from rhosocial.activerecord.backend.impl.postgresql import PostgresBackend

backend = PostgresBackend(host='127.0.0.1', user='postgres', database='app_db')
backend.connect()
dialect = backend.dialect

# VACUUM
sql, params = dialect.format_vacuum(table_name='users', analyze=True)
# → VACUUM ANALYZE "users"

# ANALYZE
sql, params = dialect.format_analyze(table_name='orders', columns=['status', 'created_at'])
# → ANALYZE "orders" ("status", "created_at")

# COMMENT ON
sql, params = dialect.format_comment(
    object_type='TABLE',
    object_name='users',
    comment='Application users'
)
# → COMMENT ON TABLE "users" IS 'Application users'

# REFRESH MATERIALIZED VIEW
sql, params = dialect.format_refresh_materialized_view(
    view_name='user_stats',
    concurrently=True
)
# → REFRESH MATERIALIZED VIEW CONCURRENTLY "user_stats"

# CREATE STATISTICS
sql, params = dialect.format_create_statistics(
    name='user_stats_obj',
    kinds=['ndistinct', 'dependencies'],
    columns=['region', 'status'],
    table_name='users'
)

Function Example

函数使用示例。

function_example.py
from rhosocial.activerecord.backend.impl.postgresql.dialect import PostgresDialect

dialect = PostgresDialect()

# JSONB 路径查询
sql, params = dialect.jsonb_path_query(
    column='metadata',
    path='$.tags[*] ? (@.type == "skill")'
)
# → jsonb_path_query("metadata", '$.tags[*] ? (@.type == "skill")')

# JSONB 包含检测（@> 操作符）
sql, params = dialect.jsonb_contains(
    column='settings',
    value='{"theme": "dark"}'
)
# → "settings" @> '{"theme": "dark"}'::jsonb

# Range 构造
sql, params = dialect.int4range(lower=1, upper=100)
# → int4range(1, 100)

# 系统函数
sql, params = dialect.pg_typeof(expression='created_at')
# → pg_typeof("created_at")

sql, params = dialect.current_setting(name='work_mem')
# → current_setting('work_mem')

# string_agg 聚合
sql, params = dialect.string_agg(expression='name', delimiter=',')
# → string_agg("name", ',')

# array_agg 聚合
sql, params = dialect.array_agg(expression='tag')
# → array_agg("tag")
Transaction

事务控制。

PostgresTransactionManager 支持全部四种 SQL 标准隔离级别。默认 READ_COMMITTED。支持 SET TRANSACTION 设置。

transaction_control.py
from rhosocial.activerecord.backend.transaction import IsolationLevel
from rhosocial.activerecord.backend.impl.postgresql.dialect import PostgresDialect

# 标准事务（默认 READ COMMITTED）
with backend.transaction():
    backend.execute("INSERT INTO users ...", ())

# 指定隔离级别
with backend.transaction(isolation_level=IsolationLevel.SERIALIZABLE):
    ...

# PostgreSQL 四种隔离级别
# READ_UNCOMMITTED — PostgreSQL 实际提升为 READ_COMMITTED
# READ_COMMITTED — 默认，每次语句看到最新快照
# REPEATABLE_READ — 事务内看到一致性快照
# SERIALIZABLE — 严格串行化，检测序列化失败

# SET TRANSACTION（事务级别配置）
dialect = PostgresDialect()
sql, params = dialect.format_set_transaction(
    isolation_level='SERIALIZABLE',
    read_only=True,
    deferrable=True
)
# → SET TRANSACTION ISOLATION LEVEL SERIALIZABLE READ ONLY DEFERRABLE

# Savepoint 支持
with backend.transaction():
    backend.execute("INSERT ...", ())
    with backend.transaction(): # SAVEPOINT
        backend.execute("UPDATE ...", ())
        # ROLLBACK TO SAVEPOINT → 自动回滚
Async

异步支持。

AsyncPostgresBackend / AsyncPostgresTransactionManager 懒加载，需安装 asyncpg。

async_postgres.py
from rhosocial.activerecord.backend.impl.postgresql import AsyncPostgresBackend
# 需安装: pip install rhosocial-activerecord-postgresql[async]

backend = AsyncPostgresBackend(
    host='127.0.0.1',
    port=5432,
    user='app_user',
    password='secret',
    database='app_db',
)
await backend.connect()

async with backend.transaction():
    result = await backend.execute("SELECT ...", ())

await backend.disconnect()

# 懒加载机制：导入同步组件不需要 asyncpg
from rhosocial.activerecord.backend.impl.postgresql import PostgresBackend # ✓ 无需 asyncpg
from rhosocial.activerecord.backend.impl.postgresql import AsyncPostgresBackend # ✗ 需要 asyncpg
